  private void DoDraggingAndSelection(Rect listRect)
  {
    UnityEngine.Event current = UnityEngine.Event.current;
    int index1 = this.index;
    bool flag = false;
    switch (current.GetTypeForControl(this.id))
    {
      case UnityEngine.EventType.MouseDown:
        if (listRect.Contains(UnityEngine.Event.current.mousePosition) && UnityEngine.Event.current.button <= 0)
        {
          EditorGUI.EndEditingActiveTextField();
          int rowIndex = this.GetRowIndex(UnityEngine.Event.current.mousePosition.y - listRect.y);
          if (EditorGUI.actionKey && this.multiSelect)
          {
            if (this.IsSelected(rowIndex))
              this.Deselect(rowIndex);
            else
              this.Select(rowIndex, true);
          }
          else if (current.shift && this.multiSelect && this.selectedIndices.Count > 0)
            this.SelectRange(rowIndex < this.selectedIndices[0] ? this.selectedIndices[this.selectedIndices.Count - 1] : this.selectedIndices[0], rowIndex);
          else
            this.Select(rowIndex);
          if (this.index >= 0 && this.m_Draggable)
          {
            float localY = UnityEngine.Event.current.mousePosition.y - listRect.y;
            this.m_DragOffset = localY - this.GetElementYOffset(this.GetRowIndex(localY));
            this.UpdateDraggedY(listRect);
            GUIUtility.hotControl = this.id;
            this.m_SlideGroup.Reset();
            this.m_NonDragTargetIndices = new List<int>();
          }
          this.GrabKeyboardFocus();
          current.Use();
          flag = true;
          break;
        }
        break;
      case UnityEngine.EventType.MouseUp:
        flag = false;
        if (!this.m_Draggable)
        {
          if (this.onMouseUpCallback != null && this.IsMouseInsideActiveElement(listRect))
          {
            this.onMouseUpCallback(this);
            break;
          }
          break;
        }
        if (GUIUtility.hotControl == this.id)
        {
          current.Use();
          try
          {
            int rowIndex = this.CalculateRowIndex(listRect);
            if (this.index != rowIndex && this.m_Dragging)
            {
              if (this.m_SerializedObject != null && this.m_Elements != null)
              {
                Undo.RegisterCompleteObjectUndo(this.m_SerializedObject.targetObjects, ReorderableList.Defaults.undoMove);
                this.m_Elements.MoveArrayElement(this.index, rowIndex);
              }
              else if (this.m_ElementList != null)
              {
                object element = this.m_ElementList[this.index];
                for (int index2 = 0; index2 < this.m_ElementList.Count - 1; ++index2)
                {
                  if (index2 >= this.index)
                    this.m_ElementList[index2] = this.m_ElementList[index2 + 1];
                }
                for (int index3 = this.m_ElementList.Count - 1; index3 > 0; --index3)
                {
                  if (index3 > rowIndex)
                    this.m_ElementList[index3] = this.m_ElementList[index3 - 1];
                }
                this.m_ElementList[rowIndex] = element;
              }
              int index4 = this.index;
              int num = rowIndex;
              int count = this.count;
              if (this.m_SerializedObject != null && this.m_Elements != null && !this.isOverMaxMultiEditLimit)
                EditorGUIUtility.MoveArrayExpandedState(this.m_Elements, index4, num);
              this.index = rowIndex;
              if (this.onReorderCallbackWithDetails != null)
              {
                this.onReorderCallbackWithDetails(this, index4, num);
              }
              else
              {
                ReorderableList.ReorderCallbackDelegate onReorderCallback = this.onReorderCallback;
                if (onReorderCallback != null)
                  onReorderCallback(this);
              }
              ReorderableList.ChangedCallbackDelegate onChangedCallback = this.onChangedCallback;
              if (onChangedCallback != null)
                onChangedCallback(this);
              GUI.changed = true;
              break;
            }
            ReorderableList.SelectCallbackDelegate onMouseUpCallback = this.onMouseUpCallback;
            if (onMouseUpCallback != null)
              onMouseUpCallback(this);
            break;
          }
          catch (ObjectDisposedException ex)
          {
            break;
          }
          finally
          {
            GUIUtility.hotControl = 0;
            this.m_NonDragTargetIndices = (List<int>) null;
            this.m_Dragging = false;
          }
        }
        else
          break;
      case UnityEngine.EventType.MouseDrag:
        if (!this.m_Draggable || GUIUtility.hotControl != this.id || current.modifiers != 0)
        {
          this.m_Dragging = false;
          break;
        }
        this.m_Dragging = true;
        if (this.m_Selection.Count > 1)
          this.index = this.GetRowIndex(UnityEngine.Event.current.mousePosition.y - listRect.y);
        ReorderableList.DragCallbackDelegate mouseDragCallback = this.onMouseDragCallback;
        if (mouseDragCallback != null)
          mouseDragCallback(this);
        this.UpdateDraggedY(listRect);
        if ((UnityEngine.Object) PropertyEditor.HoveredPropertyEditor != (UnityEngine.Object) null)
          PropertyEditor.HoveredPropertyEditor.AutoScroll(GUIUtility.GUIToScreenPoint(new Vector2(0.0f, Mathf.Clamp(current.mousePosition.y, listRect.y, listRect.y + listRect.height))) - new Vector2(0.0f, PropertyEditor.HoveredPropertyEditor.m_Pos.y));
        current.Use();
        break;
      case UnityEngine.EventType.KeyDown:
        if (((GUIUtility.keyboardControl != this.id ? 1 : (this.m_Dragging ? 1 : 0)) | (flag ? 1 : 0)) != 0)
          return;
        if (current.keyCode == KeyCode.DownArrow)
        {
          ++this.index;
          if (this.index >= this.m_Count)
            this.index = 0;
          current.Use();
        }
        if (current.keyCode == KeyCode.UpArrow)
        {
          --this.index;
          if (this.index < 0)
            this.index = this.m_Count - 1;
          current.Use();
        }
        if (current.keyCode == KeyCode.LeftArrow)
        {
          if (this.m_Elements != null)
          {
            foreach (int index5 in this.m_Selection)
            {
              if (index5 >= 0)
                this.m_Elements.GetArrayElementAtIndex(index5).isExpanded = false;
            }
          }
          ReorderableList.InvalidateParentCaches(this.m_PropertyPath);
          GUI.changed = true;
          current.Use();
        }
        if (current.keyCode == KeyCode.RightArrow)
        {
          if (this.m_Elements != null)
          {
            foreach (int selectedIndex in this.selectedIndices)
            {
              if (selectedIndex >= 0)
                this.m_Elements.GetArrayElementAtIndex(selectedIndex).isExpanded = true;
            }
          }
          ReorderableList.InvalidateParentCaches(this.m_PropertyPath);
          GUI.changed = true;
          current.Use();
        }
        if (Application.platform != RuntimePlatform.OSXEditor && current.keyCode == KeyCode.Delete || Application.platform == RuntimePlatform.OSXEditor && current.keyCode == KeyCode.Backspace && current.modifiers.HasFlag((Enum) EventModifiers.Command))
        {
          this.m_scheduleRemove = true;
          ReorderableList.InvalidateParentCaches(this.m_PropertyPath);
          current.Use();
        }
        if (current.keyCode == KeyCode.Escape && GUIUtility.hotControl == this.id)
        {
          GUIUtility.hotControl = 0;
          this.m_Dragging = false;
          current.Use();
        }
        if (current.type == UnityEngine.EventType.Used)
          this.m_Selection = this.m_Selection.Where<int>((Func<int, bool>) (i => i >= 0 && i < (this.m_Elements != null ? this.m_Elements.arraySize : this.m_ElementList.Count))).ToList<int>();
        if (index1 != this.index && (UnityEngine.Object) PropertyEditor.FocusedPropertyEditor != (UnityEngine.Object) null)
        {
          float num = 0.0f;
          if (index1 < this.index)
            num = Mathf.Min(this.GetElementHeight(this.index), PropertyEditor.FocusedPropertyEditor.scrollViewportRect.height * 0.9f);
          Vector2 vector2 = new Vector2(0.0f, listRect.y + this.GetElementYOffset(this.index) + num);
          vector2 = GUIUtility.GUIToScreenPoint(vector2) - new Vector2(0.0f, PropertyEditor.FocusedPropertyEditor.m_Pos.y);
          PropertyEditor.FocusedPropertyEditor.ScrollTo(vector2);
          break;
        }
        break;
    }
    if (!(this.index != index1 | flag) || this.onSelectCallback == null)
      return;
    this.onSelectCallback(this);
  }

  private bool IsMouseInsideActiveElement(Rect listRect)
  {
    int rowIndex = this.GetRowIndex(UnityEngine.Event.current.mousePosition.y - listRect.y);
    return this.IsSelected(rowIndex) && this.GetRowRect(rowIndex, listRect).Contains(UnityEngine.Event.current.mousePosition);
  }

  private void UpdateDraggedY(Rect listRect)
  {
    this.m_DraggedY = UnityEngine.Event.current.mousePosition.y - listRect.y;
  }

  private float GetClampedDragPosition(Rect listRect)
  {
    return Mathf.Clamp(this.m_DraggedY, this.m_DragOffset, listRect.height - this.GetElementHeight(this.index) + this.m_DragOffset);
  }

  private int CalculateRowIndex(Rect listRect)
  {
    int rowIndex = this.GetRowIndex(this.GetClampedDragPosition(listRect) - this.m_DragOffset, true);
    float num = this.m_DraggedY - this.m_DragOffset;
    if ((double) (num - this.GetElementYOffset(this.index)) < 0.0)
    {
      while (rowIndex > 0 && (double) this.GetElementYOffset(rowIndex - 1) + (double) this.GetElementHeight(rowIndex - 1) / 2.0 > (double) num)
        --rowIndex;
    }
    return rowIndex;
  }

  private int GetRowIndex(float localY, bool skipActiveElement = false)
  {
    for (int index = 0; index < this.m_Count; ++index)
    {
      float elementYoffset = this.GetElementYOffset(index);
      if (skipActiveElement)
      {
        if (index >= this.index)
          elementYoffset += (float) (-(double) this.GetElementHeight(this.index) + (double) this.GetElementHeight(index) / 2.0);
        else if (index < this.index)
          elementYoffset -= this.GetElementHeight(index) / 2f;
      }
      if ((double) elementYoffset > (double) localY)
      {
        int num;
        return num = index - 1;
      }
    }
    return this.m_Count - 1;
  }

    internal void DoAddButton(ReorderableList list, UnityEngine.Object value)
    {
      if (GUIUtility.keyboardControl != list.id)

      {

        list.serializedProperty.arraySize = list.count + 1;
        list.index = list.serializedProperty.arraySize - 1;
        if (value != (UnityEngine.Object) null)
          list.serializedProperty.GetArrayElementAtIndex(list.index).objectReferenceValue = value;
      }
      else
      {
        System.Type type1 = list.list.GetType();
        System.Type type2 = !type1.IsGenericType ? type1.GetElementType() : type1.GetTypeInfo().GenericTypeArguments[0];
        if (value != (UnityEngine.Object) null)
          list.index = list.list.Add((object) value);
        else if (type2 == typeof (string))
          list.index = list.list.Add((object) "");
        else if (list.list.GetType().GetGenericArguments()[0] != (System.Type) null)
          list.index = list.list.Add(Activator.CreateInstance(list.list.GetType().GetGenericArguments()[0]));
        else if (type2 != (System.Type) null && type2.GetConstructor(System.Type.EmptyTypes) == (ConstructorInfo) null)
          Debug.LogError((object) $"Cannot add element. Type {type2?.ToString()} has no default constructor. Implement a default constructor or implement your own add behaviour.");
        else if (type2 != (System.Type) null)
          list.index = list.list.Add(Activator.CreateInstance(type2));
        else
          Debug.LogError((object) "Cannot add element of type Null.");
      }
      Undo.SetCurrentGroupName(ReorderableList.Defaults.undoAdd);
      list.InvalidateForGUI();
    }

    public void DoRemoveButton(ReorderableList list)
    {
      if (GUIUtility.keyboardControl != list.id)
        list.GrabKeyboardFocus();
      int[] numArray1;
      if (list.m_Selection.Count <= 0)
        numArray1 = new int[1]{ list.index };
      else
        numArray1 = list.selectedIndices.Reverse<int>().ToArray<int>();
      int[] numArray2 = numArray1;
      int num = -1;
      foreach (int index1 in numArray2)
      {
        if (index1 < list.count)
        {
          if (list.serializedProperty != null)
          {
            list.serializedProperty.DeleteArrayElementAtIndex(index1);
            if (index1 < list.count - 1)
            {
              SerializedProperty serializedProperty = list.serializedProperty.GetArrayElementAtIndex(index1);
              for (int index2 = index1 + 1; index2 < list.count; ++index2)
              {
                SerializedProperty arrayElementAtIndex = list.serializedProperty.GetArrayElementAtIndex(index2);
                serializedProperty.isExpanded = arrayElementAtIndex.isExpanded;
                serializedProperty = arrayElementAtIndex;
              }
            }
          }
          else
            list.list.RemoveAt(list.index);
          num = index1;
        }
      }
      list.index = Mathf.Clamp(num - 1, 0, list.count - 1);
      Undo.SetCurrentGroupName(ReorderableList.Defaults.undoRemove);
      list.InvalidateForGUI();
    }
